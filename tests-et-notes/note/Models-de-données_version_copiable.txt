           LISTE COMPLÈTE DES
                 ENTITÉS
                                      1 User
User {
    nom:         String, required
    prenoms:     String, required
    email:       String, required, unique
    telephone:   String, optional
    mdp:         String, required
    photo:       String, optional
    role:        String, required, enum ["Admin", "Commercant", "Acheteur"]
    createdAt:   Date, default
    updatedAt:   Date, default
}




                        2 CentreCommercial
CentreCommercial {
    nom:          String, required
    description: String, optional
    adresse:      String, required,
    email: String, optional,
    telephone: String, optional,
    photo: String, optional,
    createdAt:    Date, default
    updatedAt:    Date, default
}




                                  3 PorteFeuille
PorteFeuille {
    owner:       ObjectId, required, ref "User"
    balance:     Number, required, default 0
    createdAt: Date, default
    updatedAt: Date, default
}




                             4 PFTransaction
PFTransaction {
    fromWallet: ObjectId, required, ref "PorteFeuille" // celui qui paie
    toWallet:    ObjectId, required, ref "PorteFeuille" // celui qui reçoit
    type:        String, required, enum ["Achat", "Loyer", "Commission"]
    amount:      Number, required // toujours positif
    description:String, optional
    createdAt:   Date, default
}




                                 5 Notification
Notification {
    type:        String, required, enum ["Paiement", "Achat", "Vente"]
    message:     String, required
    receveur:    ObjectId, required, ref "User"
    estLu:       Boolean, required, default false
    urlRoute:    String, optional
    createdAt:   Date, default
    updatedAt:   Date, default
}




                                      6 Etage
Etage {
    niveau: Number, required
}
                                      7 Espace
Espace {
    centreCommercial: ObjectId, required, ref "CentreCommercial"
    code:             String, required
    surface:          Number, required
    etage:            ObjectId, required, ref "Etage"
    loyer:            Number, required
    statut:           String, required, enum ["Disponible", "Occupee"]
    boutique:         ObjectId, optional, ref "Boutique"
    createdAt:        Date, default
    updatedAt:        Date, default
}




                       8 CategorieBoutique
CategorieBoutique {
    categorie: String, required
}




                                  9 Boutique
Boutique {
    nom:               String, required
    description:       String, optional
    commercant:        ObjectId, required, ref "User"
    categorie:         ObjectId, required, ref "CategorieBoutique"
    statutBoutique: String, required, enum ["Actif", "Inactif"]
    photo:             String, optional
    espace:            ObjectId, optional, ref "Espace"
    horairesHebdo: [
      {
          jour:   String, required, enum [
            "Lundi","Mardi","Mercredi","Jeudi",
            "Vendredi","Samedi","Dimanche"
          ],
          debut: String, required, // "08:00"
          fin:    String, required   // "18:00"
      }
    ],


    createdAt:         Date, default
    updatedAt:         Date, default
}




                         10 DemandeLocation
DemandeLocation {

    espace:       ObjectId, required, ref "Espace"
    etatDemande:String, required, enum ["EnAttente", "Acceptee", "Refusee"]
    createdAt:    Date, default
    updatedAt:    Date, default
}




                                  11 Recepisse

    receveur:    ObjectId, required, ref "User"

    createdAt: Date, default
}




                               12 TypeProduit
TypeProduit {

}




                                    13 Produit
Produit {
    nom:                String, required
    description:        String, optional
    prix:               Number, required
    typeProduit:        ObjectId, required, ref "TypeProduit"
    boutique:           ObjectId, required, ref "Boutique"
    tempsPreparation: String, optional // "hh:mm:ss" ou null
    stock: {
      nombreDispo:      Number, required, default 0,
      updatedAt:        Date, default
    },
    createdAt:          Date, default
    updatedAt:          Date, default
}




                                    14 Facture
Facture {
    acheteur:      ObjectId, required, ref "User"

    createdAt: Date, default
}




                                       15 Achat
Achat {
    acheteur: ObjectId, required, ref "User"
    produit:       ObjectId, required, ref "Produit"

    typeAchat: {
        type:        String, required, enum ["Recuperer", "Livrer"]
        dateDebut: Date, required
        dateFin:     Date, required
    }
    etat:          String, required, enum ["EnAttente", "Validee"]
    createdAt: Date, default
    updatedAt: Date, default
